### 내 답
<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include stdio.h

int main(void) {
	int i, j, N, t,s=0,max=0;
	int arr[500000] = { 0 }, count[500000] = { 0 };
	float avg=0;
	scanf("%d", &N);
	for (i = 0; i < N; i++) {
		scanf("%d", &t);
		avg += t;
		for (j = 0; j <= i; j++) {
			if (arr[j] == t) {
				count[j]++;
				break;
			}
			else if(j==i) {
				s++;
				arr[i] = t;
				count[i]++;
			}
		}
	}
//정렬
	for (i = 0; i < N; i++) {
		for (j = 0; j < N - 1; j++) {
			if (arr[i] < arr[j]) {
				t = arr[j];
				arr[j] = arr[i];
				arr[i] = t;
				t = count[j];
				count[j] = count[i];
				count[i] = t;
			}
		}
	}
	printf("%.0f\n", avg/N);
	printf("%d\n", arr[s/2]);
	printf("%d\n", arr[max]);
	printf("%d", arr[s-1] - arr[0]);
}
