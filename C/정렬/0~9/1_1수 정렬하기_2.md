### 내 답
<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include stdio.h

int main(void) {
	int N, i,j,t, num[1000] = { 0 };
	scanf("%d\n", &N);
	//값넣기
	for (i = 0; i < N; i++) {
		scanf("%d", &num[i]);
		for (j = i;; j--) {
			if (i == 0||j==0) {
				break;
			}
			else if (num[j-1] < num[j]) {
				break;
			}
			else {
				t = num[j-1];
				num[j-1] = num[j];
				num[j] = t;
			}
		}
	}
	for (i = 0; i < N; i++) {
			printf("%d\n", num[i]);
	}
	return 0;
}
