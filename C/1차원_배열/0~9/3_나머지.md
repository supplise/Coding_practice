#define _CRT_SECURE_NO_WARNINGS
#include stdio.h

int main(void) {
	int a, arr[10], count = 0;
	for (int i = 0; i < 10; i++) {
		scanf("%d", &a);
		arr[i] = a % 42;
	}
	for (int i = 0; i < 10; i++) {
		for (int j = 0; j < 10; j++) {
			if (arr[i] == arr[j]&&i!=j) {
				arr[i]=-1;
				break;
			}
		}
	}
	for (int i = 0; i < 10; i++) {
		if (arr[i] >= 0) {
			count++;
		}
	}
	printf("%d", count);
}
