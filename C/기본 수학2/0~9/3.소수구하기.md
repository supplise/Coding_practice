<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include stdio.h
#include string.h

int main(void) {
	int a, b;
	scanf("%d %d", &a, &b);
	for (int i = a; i <= b; i++) {
		for (int j = 2; j <= i; j++) {
			if (i % j == 0 && j != i || 2 != 0 && i % 2 != 1) {
				break;
			}
			else if (i == j) {
				printf("%d\n", i);
			}
		}
	}
}
