#define _CRT_SECURE_NO_WARNINGS
#include stdio.h
#include string.h

int main(void) {
	int a,t[300000] = { 0 , },count;
		for (int i = 2; i <= 123456*2; i++) {
			t[1] = 1;
			for (int j = i * 2; j <= 123456*2; j += i) {
				t[j] = 1;
			}
		}
		for (;;) {
			scanf("%d", &a);
			count = 0;
			if (a == 0) { break; }
			for (int i = a+1; i <= a * 2; i++) {
				if (t[i] == 0) {
					count++;
				}
			}
			printf("%d\n", count);
		}
}
