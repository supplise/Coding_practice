<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main(void) {
	int a, b, c[10000] = { 0 },t=0,i;
	scanf("%d %d", &a, &b);
	//값넣어주기
	for (i = 0; i < a; i++) {
		scanf("%d",&c[i]);
	}
	for (i=0;i<a;i++) {
		if (t > 1) {
			break;
		}
		for (int j = 0; j < a; j++) {
			if (i == j) {
				continue;
			}
			for (int k = 0; k < a; k++) {
				if (i == k || j == k) {
					continue;
				}
				if (b - c[i] == c[k] + c[j]) {
					t=c[i]+c[j]+c[k];
					break;
				}
				if (i == a - 1 && j == a - 2 && k == a-3 && b - c[i] != c[k] + c[j])
				{
					i = -1;
					b -= 1;
				}
			}
		}
	}
	printf("%d", t);
}

4.같은 값이 나올경우는 컨티뉴로 넘겨줍니다<br>
5.그리고 만약 i가 갈 수 있는 최대 j가 갈 수 있는 최대 k가 갈 수 있는 최대가 되면 i가 -1(++되기에)으로 되고 b를 -1을 해주게 만들었습니다.<br>
