#define _CRT_SECURE_NO_WARNINGS
#include stdio.h
#include string.h

int main(void) {
	char a[10000],total=0;
	scanf("%s", a);
	for (int i = 0; a[i] != '\0'; i++) {
		if (a[i]>=83&& a[i]%3 == 2|| a[i]==90) {
			total += (a[i] - 67) / 3 + 2;
		}
		else {
			total += (a[i] - 65) / 3 + 2;
		}
	}
	printf("%d", total+strlen(a));
}
