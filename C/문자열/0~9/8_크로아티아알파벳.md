<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include stdio.h
#include string.h

int main(void) {
	char a[10000],count=0;
	scanf("%s", a);
	count = strlen(a);
	for (int i = 0; a[i] != '\0'; i++) {
		if (a[i] == 'd' && a[i + 1] == 'z' && a[i + 2] == '=') {
			count--;
		}
		else if (a[i]=='=' || a[i] == '-') {
			count--;
		}
		else if (a[i] == 'l' && a[i + 1] == 'j') {
			count--;
		}
		else if (a[i] == 'n' && a[i + 1] == 'j') {
			count--;
		}
	}
	printf("%d", count);
}
