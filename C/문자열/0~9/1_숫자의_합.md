#define _CRT_SECURE_NO_WARNINGS
#include stdio.h

int main(void) {

	char b;
	scanf("%d\n", &a);

	for (int i = 0; i < a; i++) {
		scanf("%c", &b);
		total = total + ((int)b-48);
	}
	printf("%d", total);

}
