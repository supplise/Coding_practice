#include stdio.h

void main() {
		for (int i = 9; i >= 7; i--) {
			for (int j = 1; j <= 9; j++) {
				for (int k = 0; k <= 6; k += 3) {
					printf("%d * %d = %d\t", i - k, j, (i - k)*j);
				}
				printf("\n");
			}
			printf("\n");
		}
}
