### 내 답
<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main(void) {
	int count,a,b;
	scanf("%d", &count);
	for (int i = 1; i <= count; i++) {
		scanf("%d %d", &a, &b);
		printf("%d\n", a + b);
	}
}

#### 느낀점(배운점)
