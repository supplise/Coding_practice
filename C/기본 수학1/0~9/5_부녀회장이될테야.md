### 내 답
<pre><code>
#define _CRT_SECURE_NO_WARNINGS
#include stdio.h
#include string.h

int main(void) {
	int t, k, n, i,x,y;
	int a[16][16] = { 0 ,};
	scanf("%d", &t);
	//배열구간
	for (x = 0; x <= 15; x++) {
		for (y =0; y <= 15; y++) {
			if (x == 0) {
				a[x][y] = y;
			}
			else {
				for (int e = 1; e < y; e++) {
					a[x][y-1] += a[x - 1][e];
				}
			}
		}
	}
	//출력구간
	for (i = 0; i < t; i++) {
		scanf("%d %d", &k,&n);
		printf("%d\n", a[k][n]);
	}
}
