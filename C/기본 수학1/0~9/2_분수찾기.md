<code>
#define _CRT_SECURE_NO_WARNINGS
#include stdio.h
#include string.h

int main(void) {
	int t, x = 1, trig = 1, a = 1, b = 1;
	scanf("%d", &t);
	for (int i = 1; i < t; i++) {
		if (b == x && trig == 1) {
			x++;
			b++;
			trig = 0;
		}else
		if (trig == 0 && b == x || trig == 0 && a < x) {
			a++;
			b--;
		}else
		if (a == x && trig == 0) {
			x++;
			a++;
			trig = 1;
		}else
		if (trig == 1 && a == x|| trig == 1&&b<x) {
			a--;
			b++;
		}
	}
	printf("%d/%d", a, b);
}
