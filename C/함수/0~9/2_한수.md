#define _CRT_SECURE_NO_WARNINGS
#include stdio.h

int main(void) {
	int val,count=0,abs1,abs2,abs3;
	scanf("%d", &val);
	for (int i = 1; i <= val; i++) {
		if (i < 100) {
			count++;
		}
		abs1 = (i / 100 - (i / 10) % 10);
		abs2 = ((i / 10) % 10 - i % 10);
		abs3 = i / 100 - i % 10;
		if (abs1 < 0) {
			abs1 = -abs1;
		}
		if (abs2 < 0) {
			abs2 = -abs2;
		}
		if (abs3 < 0) {
			abs3 = -abs3;
		}
		for(int j=0;j<5;j++){
			if (i>=100 && abs1==j && abs2==j&&abs3==(j*2)) {
				count++;
			}
		}
	}
	printf("%d", count);
}
